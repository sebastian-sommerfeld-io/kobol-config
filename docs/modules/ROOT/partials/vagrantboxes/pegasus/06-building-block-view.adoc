@startuml

skinparam linetype ortho

skinparam ComponentBorderColor black

component vb as "Vagrantbox" <<pegasus>> {
    collections containers <<docker>>
    collections packages <<apt & snap>>
    collections wrappers <<docker>>

    packages -[hidden]right- containers
    containers -[hidden]right- wrappers
}

folder ssh as ".ssh" <<kobol>>
folder tmp <<kobol>>
folder work <<kobol>>
file gitignore as ".gitignore" <<kobol>>


ssh -up-> vb
tmp -up-> vb
work -up-> vb
gitignore -up-> vb

@enduml
----

. Folders and Config from host is mounted into Vagrantbox to make sure the host machine supplies all secrets to the Vagrantbox.
. Wrappers are scripts wrapping Docker containers to use them by their respective default command.
